243        """
244        Returns the operator which is a spin 1/2 sigma_x on the list of idxs. The states to swap are specified in states.
245

408    def PairInteraction(coords, interaction_coeff, states, atom_types = None, selection_idxs = None, powers = None):
409        """
410        Returns the interaction operator for atoms at particular coordinates with interaction coefficients. 
411
412        Args:
413            coords: A list of coordinates for the locations of each atom.
414            interaction_coeff: A list of lists of interaction coefficients. Suppose there are 3 types of atoms, then the interaction array is \
415            $ [[V_{11}, V_{12}, V_{13}], [V_{22}, V_{23}], [V_{33}]] $. A single number can also be used if there is only one type of atom.
416            states: A list of the interacting state for each atom. A single number can be used for the same state on each atom.
417            atom_types: A list of the atom types for each atom.
418            selection_idxs: A list of the atoms involved in the interaction. By default, all atoms are involved. Note that atoms that are not selected \
419            for will have their atom_types, states and coords ignored. However their entries must still be filled.
420            powers: A list of lists of the exponent of the distance dependent interaction. See interaction_coeff for the format. By default, \
421            the power is assumed to be -6, a van der Waals interaction.
422
423        Returns:
424            The interaction operator.
425
426        Raises:
427            None
428
429        """
430        # Add error checking
431        global N
432        n_atoms = len(coords)
433        if isinstance(interaction_coeff, number):
434            interaction_coeff = [[interaction_coeff]]
435        n_types = len(interaction_coeff[0])
436        if isinstance(states, number):
437            states = [states for i in range(n_atoms)]
438        if atom_types is None:
439            atom_types = [0 for i in range(n_atoms)]
440        if powers is None:
441            powers = []
442            for i in range(n_types):
443                this_list = []
444                for j in range(n_types - i):
445                    this_list.append(-6)
446                powers.append(this_list)
447        if selection_idxs is None:
448            selection_idxs = [i for i in range(n_atoms)]
449        if isinstance(powers, int):
450            prev_power = powers
451            powers = []
452            for i in range(n_types):
453                this_list = []
454                for j in range(n_types - i):
455                    this_list.append(prev_power)
456                powers.append(this_list)
457
458        space_size = np.prod(np.array(N))
459        proj_specifier = [-1 for i in range(n_atoms)]
460        res = qt.Qobj(np.zeros((space_size, space_size)), dims=[N for i in range(2)])
461        for i in range(n_atoms):
462            if i not in selection_idxs:
463                continue
464            for j in range(i):
465                if j not in selection_idxs:
466                    continue
467                pos1 = coords[i]
468                pos2 = coords[j]
469                dist = np.sqrt((pos1[0] - pos2[0])**2 + (pos1[1] - pos2[1])**2)
470                type1 = min(atom_types[j],atom_types[i])
471                type2 = max(atom_types[j],atom_types[i])
472                inter_strength = interaction_coeff[type1][type2 - type1] * dist**(powers[type1][type2 - type1])
473                this_proj_specifier = proj_specifier.copy()
474                this_proj_specifier[i] = states[i]
475                this_proj_specifier[j] = states[j]
476                res = res + inter_strength * Operators.proj(this_proj_specifier)
477        return res

481        """
482            Returns a controlled gate.
483
484            Args:
485                bit_type: A list of values for each atom. A number greater than 0 indicates that atom is a control atom, and will \

488                -2 corresponds to the first element of target_op, -3 indicates the next element and so on.

492                The controlled gate.
493
494            Raises:
495                None
496
497        """

502        op1 = id_list.copy()
503        for idx, elem in enumerate(bit_type):

505                # control bit, controlling when projector into elem.

508            elif elem <= -2:

647def CalculateFidelity(func, target_op, states = None):
648    global N
649    if states is None:
650        states = []
651        for elem in N:
652            states.append([i for i in range(elem)])
653    nstates_calculated = 1
654    for elem in states:
655        nstates_calculated = nstates_calculated * len(elem)
656    nstates = np.prod(np.array(N))
657    result_matrix = np.zeros((nstates, nstates), dtype=np.complex128)
658    ideal_result_matrix = np.zeros((nstates, nstates), dtype=np.complex128)
659    op_dims = [N for i in range(2)]
660    op_shape = (nstates, nstates)
661    idx = 0
662    for perm in product(*states):
663        init_state = States.specify(perm)
664        result_state = func(init_state)
665        result_matrix[:,idx] = np.squeeze(result_state.full())
666        ideal_result_state = target_op * init_state
667        ideal_result_matrix[:, idx] = np.squeeze(ideal_result_state.full())
668        idx = idx + 1
669    result_op = qt.Qobj(result_matrix, dims=op_dims, shape=op_shape)
670    ideal_result_op = qt.Qobj(ideal_result_matrix, dims=op_dims, shape=op_shape)
671    return (result_op.dag() * ideal_result_op).tr() / nstates_calculated, result_op, ideal_result_op
